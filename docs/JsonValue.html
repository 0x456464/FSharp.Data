  • value.InnerText - *)

( Example that works

1: match value with
2: | JsonValue.Array [info; data] ->
3:     let page, pages, total = info?page, info?pages, info?total
4:     printfn "Showing page %d of %d. Total records %d" page.AsInteger pages.AsInteger total.AsInteger
5:     for record in data do
6:       if record?value <> JsonValue.Null then
7:         printfn "%d: %f" (int record?date.AsString) (float record?value.AsString)
8: | _ -> printfn "failed"
