Introducing the provider

?

 1: #r "../bin/FSharp.Data.dll"
 2: open System.IO
 3: open FSharp.Data
 4: 
 5: let data = WorldBank.GetDataContext()
 6: 
 7: data.Countries.``United Kingdom``.Indicators.``School enrollment, tertiary (% gross)``
 8: |> Seq.maxBy fst
 9: 
10: #load "lib/FSharpChart.fsx"
11: open System
12: open Samples.FSharp.Charting
13: 
14: type WorldBank = WorldBankProvider<"World Development Indicators", Asynchronous=true>
15: WorldBank.GetDataContext()
16: 
17: let wb = WorldBank.GetDataContext()
18: 
19: let countries = 
20:  [| wb.Countries.``Arab World`` 
21:     wb.Countries.``European Union``
22:     wb.Countries.Australia
23:     wb.Countries.Brazil
24:     wb.Countries.Canada
25:     wb.Countries.Chile
26:     wb.Countries.``Czech Republic``
27:     wb.Countries.Denmark
28:     wb.Countries.France
29:     wb.Countries.Greece
30:     wb.Countries.``Low income``
31:     wb.Countries.``High income``
32:     wb.Countries.``United Kingdom``
33:     wb.Countries.``United States`` |]
34: 
35: [ for c in countries ->
36:     c.Indicators.``School enrollment, tertiary (% gross)`` ]
37: |> Async.Parallel
38: |> Async.RunSynchronously
39: |> Array.map Chart.Line
40: |> Chart.Combine
